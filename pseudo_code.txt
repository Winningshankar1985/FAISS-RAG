pdf_path="./pdf.pdf"
embeddings = OpenAIEmbeddings(model="text-embedding-3-small")

def initialise_rag(query:str="",num_results:int=0):
    if not os.path.exists(db_folder_path):
       os.makedirs(db_folder_path)
    faiss_index=os.path.join(db_folder_path,"index.faiss")
    pkl_file=os.path.join(db_folder_path,"index.pkl")
     loader=pyPDFLoader(pdf_path=pdf_path)
    docs=loader.load()
    if os.path.exists(faiss_index) and os.path.exists(pkl_file):
       DB=FAISS.load_local(
            folder_path=db_folder_path,
            embeddings=embeddings
            allow_dangerous_deserialization=True
       )   
     else:
        DB=FAISS.from_documents(
            docs,
            embeddings
        ) 
        DB.save_local(db_folder_path)
     return DB.similarity_search(search_query, k=num_results)


@tool
def web_search(query:str):
    """
    use this tool to search the web to refine user inputs along with db results.
    """
    tavily_client=TavilyClient(api_key=os.environ.get("TAVILY_API_KEY"))
    return tavily_client.search(query=query)
